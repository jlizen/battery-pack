--- .github/workflows/fuzz-nightly.yml ---
# https://github.com/rust-fuzz/cargo-fuzz
name: Fuzz (nightly)

on:
  schedule:
    - cron: "0 6 * * *"
  workflow_dispatch:
    inputs:
      duration:
        description: "Fuzz duration per target in seconds"
        default: "300"

permissions:
  contents: read

jobs:
  fuzz-nightly:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@[..] # v[..]
        with:
          persist-credentials: false
      - uses: dtolnay/rust-toolchain@nightly
      - uses: Swatinem/rust-cache@[..] # v[..]
        with:
          save-if: ${{ github.ref == 'refs/heads/main' }}
          workspaces: fuzz -> target
      - uses: dtolnay/install@cargo-fuzz
      - name: Extended fuzz run
        run: |
          duration=${{ inputs.duration || '300' }}
          for target in $(cargo fuzz list); do
            cargo fuzz run "$target" -- -max_total_time="$duration"
          done
      - name: Upload crash artifacts
        if: failure()
        uses: actions/upload-artifact@[..] # v[..]
        with:
          name: fuzz-crashes
          path: fuzz/artifacts/
--- end .github/workflows/fuzz-nightly.yml ---
